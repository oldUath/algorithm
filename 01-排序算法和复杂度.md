# 冒泡排序

````js
function bullueSort(arr){
    if(arr.length == 1){
        return arr
    }
    for(var i=0;i<arr.length-1;i++){
        for(var j=0;j<arr.length-i-1;j++){
            if(arr[j]>arr[j+1]){
                var mid = arr[j]
                arr[j] = arr[j+1]
                arr[j+1] = mid
            }
        }
    }
    return arr
}
console.log(bullueSort([5,8,1,5,6,9,0]))
````
